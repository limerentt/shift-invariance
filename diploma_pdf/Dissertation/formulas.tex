%%% Настройки формул по ГОСТ %%%
% Формулы должны быть центрированы
% Нумерация формул по схеме "глава.номер"
% Интервалы - сокращенные до 6 пт (≈ 0,5 строки), согласно требованиям по оптимизации верстки

% Настройки для центрирования формул с уменьшенными отступами
\AtBeginDocument{
  \abovedisplayskip=6pt       % Отбивка сверху сокращена до 6pt
  \abovedisplayshortskip=6pt  % Отбивка сверху для коротких формул сокращена до 6pt
  \belowdisplayskip=6pt       % Отбивка снизу сокращена до 6pt
  \belowdisplayshortskip=6pt  % Отбивка снизу для коротких формул сокращена до 6pt
}

% Выключаем дополнительный интервал после абзаца в формулах
\interdisplaylinepenalty=10000  % Предотвращает дополнительные интервалы между формулами

% Настройка нумерации формул
% Если contnumeq=0 (установлено в setup.tex), то формулы нумеруются по схеме "глава.номер"
\numberwithin{equation}{chapter} 

% Предотвращает разрыв страницы сразу после формулы
\interlinepenalty=1000              % Дополнительное наказание за разрыв страницы внутри абзаца
\predisplaypenalty=10000          % Запрещаем разрыв страницы перед формулой
\postdisplaypenalty=2000        % Более высокий штраф за разрыв после формулы

%%% Расстояние между формулами
\abovedisplayskip=6pt\relax% Отбивка над формулами
\belowdisplayskip=6pt\relax% Отбивка под формулами
\abovedisplayshortskip=6pt\relax
\belowdisplayshortskip=6pt\relax
